const phone = localStorage.getItem("driverPhone"); document.getElementById("driverPhone").innerText = phone; fetch(`/api/getBalance?phone=${phone}`) .then(r => r.json()) .then(d => { if (d.balance !== undefined) { document.getElementById("balance").innerText = d.balance + " ₾"; } else { document.getElementById("balance").innerText = "0 ₾"; } }) .catch(() => { document.getElementById("balance").innerText = "0 ₾"; });
